\begin{frame}{System Architecture \& Demo Setup}
    \textbf{Distributed Environment on Microsoft Azure}
    \begin{itemize}
        \item \textbf{Leader Node (10.0.0.4):} Read-Write. \textbf{Demo Tool:} Flask Web Dashboard to run tests.
        \item \textbf{Follower Node (10.0.0.5):} Read-Only replica. \textbf{Demo Tool:} Interactive CLI Monitor for live logs.
        \item \textbf{Setup:} PostgreSQL 16 Asynchronous Streaming Replication (via WAL).
    \end{itemize}
    \vspace{0.3cm}
    \textbf{How Schema Fields Help Us Track Replication:}
    \begin{enumerate}
        \item \textbf{version:} Increments on UPDATE (Shows version rollbacks during stale reads).
        \item \textbf{last\_updated:} Microsecond timestamp (Used for lag timing).
        \item \textbf{operation\_id:} UUID v4 (Links writes between Leader and Follower).
    \end{enumerate}
\end{frame}

\begin{frame}{Exp 1: Eventual Consistency \& Latency}
    \textbf{Goal:} Measure replication lag.
    \begin{itemize}
        \item \textbf{Test A (CLI Script):} Average lag was only $\sim$ \textbf{2.50 ms}. Shows raw database speed.
        \item \textbf{Test B (Web Dashboard):} Average lag jumped to \textbf{20 ms - 40 ms}.
    \end{itemize}
    \vspace{0.2cm}
    \textbf{Why the latency gap in the web dashboard?} 
    \begin{itemize}
        \item Connection overhead: Opening/closing TCP database connections.
        \item OS Thread Jitter: Python's \texttt{time.sleep(0.001)} waking up late.
        \item Python GIL contention between web request and polling threads.
    \end{itemize}
\end{frame}

\begin{frame}{Exp 2: Monotonic Reads}
    \textbf{Goal:} Test for stale reads (version rollbacks) when hitting different nodes.
    \begin{itemize}
        \item \textbf{Test A (Follower Only):} Reading only from Follower $\rightarrow$ \textbf{0 stale reads}. Since updates apply sequentially, versions never go backward.
        \item \textbf{Test B (Leader then Follower):} Read latest from Leader, then read from Follower $\rightarrow$ \textbf{7 stale reads caught} during rapid writes.
    \end{itemize}
    \vspace{0.2cm}
    \textbf{Timing Analysis from Logs (Measuring Replication Delay):}
    \begin{itemize}
        \item \textbf{First Stale Read (v4):} Leader showed v4, Follower returned stale v3. Resolved 5ms later.
        \item \textbf{Stale Reads at v11:} Follower returned v10 across two consecutive 5ms polls. Shows a lag of 5--11ms.
    \end{itemize}
\end{frame}

\begin{frame}{Exp 3: Read-After-Write Consistency}
    \textbf{Goal:} Make sure a user sees their own writes immediately.
    \begin{itemize}
        \item \textbf{Reading from Leader:} 100\% immediate visibility (0 lag issues). Reading back from the same node is instant.
        \item \textbf{Reading from Follower:} 80\% of immediate Follower reads missed the write on the first try. Average lag: 3.30 ms.
        \item \textbf{Takeaway:} Standby replicas cannot guarantee immediate visibility of your own writes without write-redirection or pinning the user session to the primary.
    \end{itemize}
\end{frame}

\begin{frame}{Exp 4: Concurrent Writes \& Race Conditions}
    \textbf{Goal:} Check write ordering and test for booking race conditions.
    \begin{itemize}
        \item \textbf{Part 1 (Write Ordering):} 10 parallel threads run at the same time $\rightarrow$ \textbf{0 ordering issues}. The Follower replayed the WAL stream in the exact same order as Leader commits.
        \item \textbf{Part 2 (Double Booking):} 2 parallel client threads try to book the same seat at the same time $\rightarrow$ \textbf{Both succeeded (Double Booking anomaly)}.
        \item \textbf{Fix:} Needs database-level row locking (\texttt{SELECT FOR UPDATE}) or a \texttt{UNIQUE} index on the seat.
    \end{itemize}
\end{frame}

\begin{frame}{Conclusion}
    \begin{itemize}
        \item Distributed setups show the real impact of network and OS delay.
        \item Asynchronous replication is fast but means readers can get stale data from followers.
        \item \textbf{Key Takeaway:} Python connection overhead, GIL, and OS thread sleep jitter added more lag in our tests than the database replication itself.
    \end{itemize}
